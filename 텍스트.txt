    <script>
        document.addEventListener('DOMContentLoaded', function() {
            // DOM 원소 매핑 변수 선언
            const inputName = document.getElementById('input-name');
            const inputRace = document.getElementById('input-race');
            const inputJob = document.getElementById('input-job');
            const inputSkill = document.getElementById('input-skill');
            const inputNumber = document.getElementById('input-number');

            const viewName = document.getElementById('view-name');
            const viewRace = document.getElementById('view-race');
            const viewJob = document.getElementById('view-job');
            const viewSkill = document.getElementById('view-skill');
            const viewNumber = document.getElementById('view-number');

            const btnTriggerIssue = document.getElementById('btn-trigger-issue');
            const btnDownloadPng = document.getElementById('btn-download-png');
            const btnResetForm = document.getElementById('btn-reset-form');

            // 기본 플레이스홀더 데이터 셋업
            const defaults = {
                name: '홍길동',
                race: '호족',
                job: '무사',
                skill: '축지법',
                number: '虎-001'
            };

            // 가변 폰트 사이즈 제어 규칙 함수
            function adjustFontSize(element, textLength) {
                if (textLength >= 12) {
                    element.style.fontSize = '0.95rem';
                } else if (textLength >= 8) {
                    element.style.fontSize = '1.1rem';
                } else if (textLength >= 5) {
                    element.style.fontSize = '1.25rem';
                } else {
                    element.style.fontSize = '1.45rem';
                }
            }

            // 공통 컴포넌트 실시간 업데이트 프로세스 모듈
            function updateField(inputEl, viewEl, defaultText) {
                let val = inputEl.value.trim();
                if (val === '') {
                    viewEl.textContent = defaultText;
                } else {
                    viewEl.textContent = val;
                }
                adjustFontSize(viewEl, viewEl.textContent.length);
                
                // 시각적 리액션 애니메이션 부여
                viewEl.classList.remove('pulse-update');
                void viewEl.offsetWidth; // 브라우저 리플로우 강제 트리거
                viewEl.classList.add('pulse-update');
            }

            // 고유 식별 번호 자동 랜덤 생성 연산 로직
            function generateRandomSerial() {
                const randomNum = Math.floor(Math.random() * 999) + 1;
                const paddedNum = String(randomNum).padStart(3, '0');
                return `虎-${paddedNum}`;
            }

            // 실시간 이벤트 리스너 바인딩 파트
            inputName.addEventListener('input', () => updateField(inputName, viewName, defaults.name));
            inputRace.addEventListener('input', () => updateField(inputRace, viewRace, defaults.race));
            inputJob.addEventListener('input', () => updateField(inputJob, viewJob, defaults.job));
            inputSkill.addEventListener('input', () => updateField(inputSkill, viewSkill, defaults.skill));
            
            inputNumber.addEventListener('input', function() {
                let val = inputNumber.value.trim();
                if (val === '') {
                    viewNumber.textContent = defaults.number;
                } else {
                    viewNumber.textContent = val;
                }
            });

            // [발급하기] 버튼 수동 트리거 동작 처리
            btnTriggerIssue.addEventListener('click', function() {
                // 번호 칸이 공백일 경우에 한해 즉시 랜덤 난수 발급 체계 가동
                if (inputNumber.value.trim() === '') {
                    const newSerial = generateRandomSerial();
                    viewNumber.textContent = newSerial;
                    inputNumber.value = newSerial;
                }
                
                // 모든 필드 새로고침 상태 갱신
                updateField(inputName, viewName, defaults.name);
                updateField(inputRace, viewRace, defaults.race);
                updateField(inputJob, viewJob, defaults.job);
                updateField(inputSkill, viewSkill, defaults.skill);

                alert('호왕연 목패가 성공적으로 인쇄/조정되었습니다. 우측 미리보기를 확인하십시오.');
            });

            // [새로 발급하기] 데이터 초기화 동작 처리
            btnResetForm.addEventListener('click', function() {
                inputName.value = '';
                inputRace.value = '';
                inputJob.value = '';
                inputSkill.value = '';
                inputNumber.value = '';

                viewName.textContent = defaults.name;
                viewRace.textContent = defaults.race;
                viewJob.textContent = defaults.job;
                viewSkill.textContent = defaults.skill;
                viewNumber.textContent = defaults.number;

                // 폰트 크기 상태 표준 복구
                const views = [viewName, viewRace, viewJob, viewSkill];
                views.forEach(v => v.style.fontSize = '1.45rem');

                alert('입력 양식이 초기화되었습니다. 새로운 정보를 기입해 주십시오.');
            });

            // [PNG 저장] 모듈: 고화질 스케일링 기법 적용 컴포넌트 렌더러
            btnDownloadPng.addEventListener('click', function() {
                const targetZone = document.getElementById('capture-zone');
                
                // 고해상도 출력을 위한 캔버스 스케일 배율 파라미터 정의 (디바이스 대응)
                const captureScale = 3; 

                // 외부 리소스 파일 로드 시간 안전 확보 유도 후 프로세스 시동
                html2canvas(targetZone, {
                    scale: captureScale,
                    useCORS: true,
                    allowTaint: true,
                    backgroundColor: '#050505',
                    logging: false
                }).then(canvas => {
                    const dataUrl = canvas.toDataURL('image/png', 1.0);
                    const downloadLink = document.createElement('a');
                    
                    // 파일 네이밍 매핑 (식별값 연동)
                    const currentName = inputName.value.trim() || defaults.name;
                    downloadLink.download = `호왕연_목패_${currentName}.png`;
                    downloadLink.href = dataUrl;
                    
                    document.body.appendChild(downloadLink);
                    downloadLink.click();
                    document.body.removeChild(downloadLink);
                }).catch(err => {
                    console.error('목패 캡쳐 엔진에 오류가 발생했습니다:', err);
                    alert('목패 이미지 저장 중 오류가 발생했습니다. 로컬 서버 혹은 유효 도메인 환경 내 구동을 권장합니다.');
                });
            });

            // 초기 폰트 크기 자율 할당 검증 수행
            adjustFontSize(viewName, viewName.textContent.length);
            adjustFontSize(viewRace, viewRace.textContent.length);
            adjustFontSize(viewJob, viewJob.textContent.length);
            adjustFontSize(viewSkill, viewSkill.textContent.length);
        });
    </script>
</body>
